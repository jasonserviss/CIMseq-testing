p2 <- synthetic.mul %>%
  mutate(dpois = map2_dbl(real.multiplet, value, ~dpois(.x, .y, TRUE))) %>%
  mutate(cost = if_else(is.infinite(dpois), -323.0052, dpois)) %>%
  mutate(cost = cost * -1) %>%
  ggplot() + 
  geom_histogram(
    aes(value, ..ndensity..), 
    binwidth = 10, fill = "darkgrey"
  ) + 
  geom_line(
    aes(x = 0, y = 0, linetype = "Synthetic multiplet values  ")
  ) +
  facet_grid(gene ~ type, scales = "free_y") +
  geom_smooth(
     data = . %>% 
      select(value, gene, cost, type) %>% 
      distinct() %>% 
      group_by(gene, type) %>% 
      mutate(cost.norm = cost / sum(cost)),
    aes(value, cost.norm, group = interaction(gene, type), linetype = "Costs  "), 
    se = FALSE, size = 0.5, span = 0.99, n = 100, method = "loess",
    formula = y ~ x, colour = "#3366FF"
  ) +
  geom_segment(
    data = . %>% select(gene, real.multiplet) %>% distinct(),
    aes(
      x = real.multiplet, xend = real.multiplet, y = 0, yend = 1.05,
      linetype = "Real multiplet values  "
    ),
    size = 0.5, colour = "red"
  ) +
  geom_text(

      select(value, gene, cost, type) %>% 
      distinct() %>% 
      group_by(gene, type) %>%

p2 <- synthetic.mul %>%
  mutate(pos = list(seq(0, max(value) + 200, 1))) %>%
  mutate(rpois = map2(value, pos, ~dpois(.y, .x))) %>%
  #mutate(rpois = map(value, ~rpois(100, .x))) %>%
  mutate(dpois.real = map2_dbl(real.multiplet, value, ~dpois(.x, .y, TRUE))) %>%
  mutate(cost.real = if_else(is.infinite(dpois.real), -323.0052, dpois.real)) %>%
  mutate(cost.real = cost.real * -1) %>%
  mutate(dpois.lambdas = list(seq(0, max(value), length.out = 100))) %>%
  mutate(dpois = map2(real.multiplet, dpois.lambdas, function(r, v) {
    dpois(r, v, TRUE)
  })) %>%
  mutate(cost = map(dpois, function(d) {
    d <- if_else(is.infinite(d), -323.0052, d)
    d * -1
  })) %>%
  ggplot() + 
  geom_histogram(
    aes(value, ..density..), 
    binwidth = 1, fill = "black"
  ) +
  #this just facilitates the histogram legend
  geom_line(
    aes(x = 0, y = 0, linetype = "Synthetic multiplet values  ")
  ) +
  geom_line(
    data = . %>%
      select(pos, rpois, gene, type, value) %>%
      unnest() %>%
      group_by(gene, type) %>%
      mutate(rpois = (rpois - min(rpois)) / (max(rpois) - min(rpois))),
    aes(pos, rpois, group = interaction(gene, type, value)),
    size = 0.1, colour = "lightgrey"
  ) +
  facet_grid(gene ~ type, scales = "free_y") +
  geom_smooth(
     data = . %>% 
      select(dpois.lambdas, gene, cost, type) %>%
       unnest() %>%
      distinct() %>% 
      group_by(gene, type) %>% 
      #mutate(cost.norm = cost / sum(cost)),
      mutate(cost.norm = (cost - min(cost)) / (max(cost) - min(cost))),
    aes(dpois.lambdas, cost.norm, group = interaction(gene, type), linetype = "Costs  "), 
    se = FALSE, size = 0.5, span = 0.99, n = 100, method = "loess",
    formula = y ~ x, colour = "#3366FF"
  ) +
  geom_segment(
    data = . %>% select(gene, real.multiplet) %>% distinct(),
    aes(
      x = real.multiplet, xend = real.multiplet, y = 0, yend = 1.05,
      linetype = "Real multiplet values  "
    ),
    size = 0.5, colour = "red"
  ) +
  geom_text(
    data = . %>%
      select(value, gene, cost.real, type) %>% 
      distinct() %>% 
      group_by(gene, type) %>% 
      summarize(mean = mean(cost.real)),
    aes(
      x = 0, y = 1.1, label = paste0("Mean cost: ", round(mean))
    ), colour = "gray13", hjust = 0, size = 3.5
  ) +
  theme_few() +
  labs(y = "Density", x = "CPM") +
  scale_linetype_manual(values = c(1, 2, 1)) +
  guides(linetype = guide_legend(
      title = "",
      override.aes = list(
        fill = rep("white", 3),
        colour = c("#3366FF", "red", "black")
      )
  )) +
  theme(
    legend.position = "top",
    legend.key = element_blank(),
    legend.title = element_blank()
  )
